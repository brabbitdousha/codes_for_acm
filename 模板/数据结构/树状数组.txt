const int MaxN=50005;
int n,m;
int a[MaxN];
int c[MaxN];
int lowbit(int x)
{
	return x&(-x);
}
void updata(int i,int k)
{
	while(i<=n)
	{
		c[i]+=k;
		i+=lowbit(i);
	}
}
int getsum(int i)
{
	int res=0;
	while(i>0)
	{
		res+=c[i];
		i-=lowbit(i);
	}
	return res;
}
