#include<bits/stdc++.h>
using namespace std;
int T,n,m;
void topsoft();
int en=0;
int head[30007];
int ans[30007];
int in[30007];
struct node
{
	int to,next;
}e[100007];
void add(int x,int y)
{
	en++;
	e[en].to=y;
	e[en].next=head[x];
	head[x]=en;
}
int main()
{   
	scanf("%d",&T);
	while(T--)
	{
		en=0;
		memset(head,0,sizeof head);
		memset(ans,0,sizeof ans);
		memset(in,0,sizeof in);
		scanf("%d %d",&n,&m);
		int a,b;
		while(m--)
		{
			scanf("%d %d",&a,&b);
			add(b,a);
			in[a]++;
		}
		topsoft();
		printf("\n");
	}
	return 0;
}
void topsoft()
{   int ct=0;
	priority_queue<int> que;
	for(int i=1;i<=n;i++)
	if(!in[i]) que.push(i);
	while(!que.empty())
	{
		int now=que.top();
		ans[++ct]=now;
		que.pop();
		int i=head[now];
		while(i)
		{
			in[e[i].to]--;
			if(in[e[i].to]==0)
			que.push(e[i].to);
			i=e[i].next;
		}
	}

	if(i!=1){printf("%d ",ans[i]);}
