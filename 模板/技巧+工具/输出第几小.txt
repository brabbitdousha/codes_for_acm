#include<bits/stdc++.h>
using namespace std;
struct int2{
    int x;
    bool operator <(const int2& a)const{return x>a.x;}
};
priority_queue<int> pq1;
priority_queue<int2> pq2;
int a[1000000],b[1000000];
int main(){
    int n,m;
    scanf("%d%d",&n,&m);
    for (int i=0;i<n;i++)
        scanf("%d",a+i);
    for (int i=0;i<m;i++){
        int x;
        scanf("%d",&x);
        b[x-1]++;
    }
    for (int i=0;i<n;i++){
        if (pq1.empty()||(pq1.top()<a[i])) pq2.push((int2){a[i]});
        else{
            int x=pq1.top();
            pq1.pop();
            pq1.push(a[i]);
            pq2.push((int2){x});
        }
        while(b[i]){
            int x=pq2.top().x;pq2.pop();
            printf("%d\n",x);
            pq1.push(x);
            b[i]--;
        }
    }
    return 0;
}
