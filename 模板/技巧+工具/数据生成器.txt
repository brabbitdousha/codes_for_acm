#include<bits/stdc++.h>
using namespace std;
int main()
{
	FILE *fp1;
	 srand( (unsigned)time( NULL ) );
    fp1=fopen("data5.txt","w");
    fprintf(fp1,"1000\n");
    for(int i=1000;i>=1;i--)
	{ int num1 = rand() % 100 + 1;
	  int num2= rand() % 100 + 1;
	  if(num1%2==0) num1=-num1;
	  if(rand()%2==0) num2=-num2;
	 fprintf(fp1,"%d %d\n",num1,num2);
    }
    fclose(fp1);
    return 0;
}
